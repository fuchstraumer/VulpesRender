364 void Configurations::Parser::ignoreComments(std::string* line) {
365  std::size_t foundAt = 0;
366  std::size_t quotesStart = line->find("\"");
367  std::size_t quotesEnd = std::string::npos;
368  if (quotesStart != std::string::npos) {
369  quotesEnd = line->find("\"", quotesStart + 1);
370  while (quotesEnd != std::string::npos && line->at(quotesEnd - 1) == '\\') {
371  // Do not erase slash yet - we will erase it in parseLine(..) while loop
372  quotesEnd = line->find("\"", quotesEnd + 2);
373  }
374  }
375  if ((foundAt = line->find(base::consts::kConfigurationComment)) != std::string::npos) {
376  if (foundAt < quotesEnd) {
377  foundAt = line->find(base::consts::kConfigurationComment, quotesEnd + 1);
378  }
379  *line = line->substr(0, foundAt);
380  }

399 bool Configurations::Parser::parseLine(std::string* line, std::string* currConfigStr, std::string* currLevelStr,
400  Level* currLevel,
401  Configurations* conf) {
402  ConfigurationType currConfig = ConfigurationType::Unknown;
403  std::string currValue = std::string();
404  *line = base::utils::Str::trim(*line);
405  if (isComment(*line)) return true;
406  ignoreComments(line);
407  *line = base::utils::Str::trim(*line);
408  if (line->empty()) {
409  // Comment ignored
410  return true;
411  }
412  if (isLevel(*line)) {
413  if (line->size() <= 2) {
414  return true;
415  }
416  *currLevelStr = line->substr(1, line->size() - 2);
417  *currLevelStr = base::utils::Str::toUpper(*currLevelStr);
418  *currLevelStr = base::utils::Str::trim(*currLevelStr);
419  *currLevel = LevelHelper::convertFromString(currLevelStr->c_str());
420  return true;
421  }
422  if (isConfig(*line)) {
423  std::size_t assignment = line->find('=');
424  *currConfigStr = line->substr(0, assignment);
425  *currConfigStr = base::utils::Str::toUpper(*currConfigStr);
426  *currConfigStr = base::utils::Str::trim(*currConfigStr);
427  currConfig = ConfigurationTypeHelper::convertFromString(currConfigStr->c_str());
428  currValue = line->substr(assignment + 1);
429  currValue = base::utils::Str::trim(currValue);
430  std::size_t quotesStart = currValue.find("\"", 0);
431  std::size_t quotesEnd = std::string::npos;
432  if (quotesStart != std::string::npos) {
433  quotesEnd = currValue.find("\"", quotesStart + 1);
434  while (quotesEnd != std::string::npos && currValue.at(quotesEnd - 1) == '\\') {
435  currValue = currValue.erase(quotesEnd - 1, 1);
436  quotesEnd = currValue.find("\"", quotesEnd + 2);
437  }
438  }
439  if (quotesStart != std::string::npos && quotesEnd != std::string::npos) {
440  // Quote provided - check and strip if valid
441  ELPP_ASSERT((quotesStart < quotesEnd), "Configuration error - No ending quote found in ["
442  << currConfigStr << "]");
443  ELPP_ASSERT((quotesStart + 1 != quotesEnd), "Empty configuration value for [" << currConfigStr << "]");
444  if ((quotesStart != quotesEnd) && (quotesStart + 1 != quotesEnd)) {
445  // Explicit check in case if assertion is disabled
446  currValue = currValue.substr(quotesStart + 1, quotesEnd - 1);
447  }
448  }
449  }
450  ELPP_ASSERT(*currLevel != Level::Unknown, "Unrecognized severity level [" << *currLevelStr << "]");
451  ELPP_ASSERT(currConfig != ConfigurationType::Unknown, "Unrecognized configuration [" << *currConfigStr << "]");
452  if (*currLevel == Level::Unknown || currConfig == ConfigurationType::Unknown) {
453  return false; // unrecognizable level or config
454  }
455  conf->set(*currLevel, currConfig, currValue);
456  return true;
457 }

783  while (*pattern) {
784  switch (*pattern) {
785  case '?':
786  if (!*str)
787  return false;
788  ++str;
789  ++pattern;
790  break;
791  case '*':
792  if (wildCardMatch(str, pattern + 1))
793  return true;
794  if (*str && wildCardMatch(str + 1, pattern))
795  return true;
796  return false;

799  return false;
800  break;
801  }
802  }

1174 char* DateTime::parseFormat(char* buf, std::size_t bufSz, const char* format, const struct tm* tInfo,
1175  std::size_t msec, const base::SubsecondPrecision* ssPrec) {
1176  const char* bufLim = buf + bufSz;
1177  for (; *format; ++format) {
1178  if (*format == base::consts::kFormatSpecifierChar) {
1179  switch (*++format) {
1180  case base::consts::kFormatSpecifierChar: // Escape
1181  break;
1182  case '\0': // End
1183  --format;
1184  break;
1185  case 'd': // Day
1186  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_mday, 2, buf, bufLim);
1187  continue;
1188  case 'a': // Day of week (short)
1189  buf = base::utils::Str::addToBuff(base::consts::kDaysAbbrev[tInfo->tm_wday], buf, bufLim);
1190  continue;
1191  case 'A': // Day of week (long)
1192  buf = base::utils::Str::addToBuff(base::consts::kDays[tInfo->tm_wday], buf, bufLim);
1193  continue;
1194  case 'M': // month
1195  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_mon + 1, 2, buf, bufLim);
1196  continue;
1197  case 'b': // month (short)
1198  buf = base::utils::Str::addToBuff(base::consts::kMonthsAbbrev[tInfo->tm_mon], buf, bufLim);
1199  continue;
1200  case 'B': // month (long)
1201  buf = base::utils::Str::addToBuff(base::consts::kMonths[tInfo->tm_mon], buf, bufLim);
1202  continue;
1203  case 'y': // year (two digits)
1204  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_year + base::consts::kYearBase, 2, buf, bufLim);
1205  continue;
1206  case 'Y': // year (four digits)
1207  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_year + base::consts::kYearBase, 4, buf, bufLim);
1208  continue;
1209  case 'h': // hour (12-hour)
1210  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_hour % 12, 2, buf, bufLim);
1211  continue;
1212  case 'H': // hour (24-hour)
1213  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_hour, 2, buf, bufLim);
1214  continue;
1215  case 'm': // minute
1216  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_min, 2, buf, bufLim);
1217  continue;
1218  case 's': // second
1219  buf = base::utils::Str::convertAndAddToBuff(tInfo->tm_sec, 2, buf, bufLim);
1220  continue;
1221  case 'z': // subsecond part
1222  case 'g':
1223  buf = base::utils::Str::convertAndAddToBuff(msec, ssPrec->m_width, buf, bufLim);
1224  continue;
1225  case 'F': // AM/PM
1226  buf = base::utils::Str::addToBuff((tInfo->tm_hour >= 12) ? base::consts::kPm : base::consts::kAm, buf, bufLim);
1227  continue;
1228  default:
1229  continue;
1230  }
1231  }
1232  if (buf == bufLim) break;
1233  *buf++ = *format;
1234  }
1235  return buf;
1236 }

1408 void LogFormat::parseFromFormat(const base::type::string_t& userFormat) {
1409  // We make copy because we will be changing the format
1410  // i.e, removing user provided date format from original format
1411  // and then storing it.
1412  base::type::string_t formatCopy = userFormat;
1413  m_flags = 0x0;
1414  auto conditionalAddFlag = [&](const base::type::char_t* specifier, base::FormatFlags flag) {
1415  std::size_t foundAt = base::type::string_t::npos;
1416  while ((foundAt = formatCopy.find(specifier, foundAt + 1)) != base::type::string_t::npos) {
1417  if (foundAt > 0 && formatCopy[foundAt - 1] == base::consts::kFormatSpecifierChar) {
1418  if (hasFlag(flag)) {
1419  // If we already have flag we remove the escape chars so that '%%' is turned to '%'
1420  // even after specifier resolution - this is because we only replaceFirst specifier
1421  formatCopy.erase(foundAt > 0 ? foundAt - 1 : 0, 1);
1422  ++foundAt;
1423  }
1424  } else {
1425  if (!hasFlag(flag)) addFlag(flag);
1426  }
1427  }
1428  };
1429  conditionalAddFlag(base::consts::kAppNameFormatSpecifier, base::FormatFlags::AppName);
1430  conditionalAddFlag(base::consts::kSeverityLevelFormatSpecifier, base::FormatFlags::Level);
1431  conditionalAddFlag(base::consts::kSeverityLevelShortFormatSpecifier, base::FormatFlags::LevelShort);
1432  conditionalAddFlag(base::consts::kLoggerIdFormatSpecifier, base::FormatFlags::LoggerId);
1433  conditionalAddFlag(base::consts::kThreadIdFormatSpecifier, base::FormatFlags::ThreadId);
1434  conditionalAddFlag(base::consts::kLogFileFormatSpecifier, base::FormatFlags::File);
1435  conditionalAddFlag(base::consts::kLogFileBaseFormatSpecifier, base::FormatFlags::FileBase);
1436  conditionalAddFlag(base::consts::kLogLineFormatSpecifier, base::FormatFlags::Line);
1437  conditionalAddFlag(base::consts::kLogLocationFormatSpecifier, base::FormatFlags::Location);
1438  conditionalAddFlag(base::consts::kLogFunctionFormatSpecifier, base::FormatFlags::Function);
1439  conditionalAddFlag(base::consts::kCurrentUserFormatSpecifier, base::FormatFlags::User);
1440  conditionalAddFlag(base::consts::kCurrentHostFormatSpecifier, base::FormatFlags::Host);
1441  conditionalAddFlag(base::consts::kMessageFormatSpecifier, base::FormatFlags::LogMessage);
1442  conditionalAddFlag(base::consts::kVerboseLevelFormatSpecifier, base::FormatFlags::VerboseLevel);
1443  // For date/time we need to extract user's date format first
1444  std::size_t dateIndex = std::string::npos;
1445  if ((dateIndex = formatCopy.find(base::consts::kDateTimeFormatSpecifier)) != std::string::npos) {
1446  while (dateIndex > 0 && formatCopy[dateIndex - 1] == base::consts::kFormatSpecifierChar) {
1447  dateIndex = formatCopy.find(base::consts::kDateTimeFormatSpecifier, dateIndex + 1);
1448  }
1449  if (dateIndex != std::string::npos) {
1450  addFlag(base::FormatFlags::DateTime);
1451  updateDateFormat(dateIndex, formatCopy);
1452  }
1453  }
1454  m_format = formatCopy;
1455  updateFormatSpec();
1456 }

1593 void TypedConfigurations::build(Configurations* configurations) {
1594  base::threading::ScopedLock scopedLock(lock());
1595  auto getBool = [] (std::string boolStr) -> bool { // Pass by value for trimming
1596  base::utils::Str::trim(boolStr);
1597  return (boolStr == "TRUE" || boolStr == "true" || boolStr == "1");
1598  };
1599  std::vector<Configuration*> withFileSizeLimit;
1600  for (Configurations::const_iterator it = configurations->begin(); it != configurations->end(); ++it) {
1601  Configuration* conf = *it;
1602  // We cannot use switch on strong enums because Intel C++ dont support them yet
1603  if (conf->configurationType() == ConfigurationType::Enabled) {
1604  setValue(conf->level(), getBool(conf->value()), &m_enabledMap);
1605  } else if (conf->configurationType() == ConfigurationType::ToFile) {
1606  setValue(conf->level(), getBool(conf->value()), &m_toFileMap);
1607  } else if (conf->configurationType() == ConfigurationType::ToStandardOutput) {
1608  setValue(conf->level(), getBool(conf->value()), &m_toStandardOutputMap);
1609  } else if (conf->configurationType() == ConfigurationType::Filename) {
1610  // We do not yet configure filename but we will configure in another
1611  // loop. This is because if file cannot be created, we will force ToFile
1612  // to be false. Because configuring logger is not necessarily performance
1613  // sensative operation, we can live with another loop; (by the way this loop
1614  // is not very heavy either)
1615  } else if (conf->configurationType() == ConfigurationType::Format) {
1616  setValue(conf->level(), base::LogFormat(conf->level(),
1617  base::type::string_t(conf->value().begin(), conf->value().end())), &m_logFormatMap);
1618  } else if (conf->configurationType() == ConfigurationType::SubsecondPrecision) {
1619  setValue(Level::Global,
1620  base::SubsecondPrecision(static_cast<int>(getULong(conf->value()))), &m_subsecondPrecisionMap);
1621  } else if (conf->configurationType() == ConfigurationType::PerformanceTracking) {
1622  setValue(Level::Global, getBool(conf->value()), &m_performanceTrackingMap);
1623  } else if (conf->configurationType() == ConfigurationType::MaxLogFileSize) {
1624  setValue(conf->level(), static_cast<std::size_t>(getULong(conf->value())), &m_maxLogFileSizeMap);
1625 #if !defined(ELPP_NO_DEFAULT_LOG_FILE)
1626  withFileSizeLimit.push_back(conf);
1627 #endif // !defined(ELPP_NO_DEFAULT_LOG_FILE)
1628  } else if (conf->configurationType() == ConfigurationType::LogFlushThreshold) {
1629  setValue(conf->level(), static_cast<std::size_t>(getULong(conf->value())), &m_logFlushThresholdMap);
1630  }
1631  }
1632  // As mentioned earlier, we will now set filename configuration in separate loop to deal with non-existent files
1633  for (Configurations::const_iterator it = configurations->begin(); it != configurations->end(); ++it) {
1634  Configuration* conf = *it;
1635  if (conf->configurationType() == ConfigurationType::Filename) {
1636  insertFile(conf->level(), conf->value());
1637  }
1638  }
1639  for (std::vector<Configuration*>::iterator conf = withFileSizeLimit.begin();
1640  conf != withFileSizeLimit.end(); ++conf) {
1641  // This is not unsafe as mutex is locked in currect scope
1642  unsafeValidateFileRolling((*conf)->level(), base::defaultPreRollOutCallback);
1643  }
1644 }

1872 void VRegistry::setModules(const char* modules) {
1873  base::threading::ScopedLock scopedLock(lock());
1874  auto addSuffix = [](std::stringstream& ss, const char* sfx, const char* prev) {
1875  if (prev != nullptr && base::utils::Str::endsWith(ss.str(), std::string(prev))) {
1876  std::string chr(ss.str().substr(0, ss.str().size() - strlen(prev)));
1877  ss.str(std::string(""));
1878  ss << chr;
1879  }
1880  if (base::utils::Str::endsWith(ss.str(), std::string(sfx))) {
1881  std::string chr(ss.str().substr(0, ss.str().size() - strlen(sfx)));
1882  ss.str(std::string(""));
1883  ss << chr;
1884  }
1885  ss << sfx;
1886  };

1888  if (!base::utils::hasFlag(LoggingFlag::DisableVModulesExtensions, *m_pFlags)) {
1889  addSuffix(ss, ".h", nullptr);
1890  m_modules.insert(std::make_pair(ss.str(), level));
1891  addSuffix(ss, ".c", ".h");
1892  m_modules.insert(std::make_pair(ss.str(), level));
1893  addSuffix(ss, ".cpp", ".c");
1894  m_modules.insert(std::make_pair(ss.str(), level));
1895  addSuffix(ss, ".cc", ".cpp");
1896  m_modules.insert(std::make_pair(ss.str(), level));
1897  addSuffix(ss, ".cxx", ".cc");
1898  m_modules.insert(std::make_pair(ss.str(), level));
1899  addSuffix(ss, ".-inl.h", ".cxx");
1900  m_modules.insert(std::make_pair(ss.str(), level));
1901  addSuffix(ss, ".hxx", ".-inl.h");
1902  m_modules.insert(std::make_pair(ss.str(), level));
1903  addSuffix(ss, ".hpp", ".hxx");
1904  m_modules.insert(std::make_pair(ss.str(), level));
1905  addSuffix(ss, ".hh", ".hpp");
1906  }
1907  m_modules.insert(std::make_pair(ss.str(), level));
1908  };
1909  bool isMod = true;
1910  bool isLevel = false;
1911  std::stringstream ss;
1912  int level = -1;
1913  for (; *modules; ++modules) {
1914  switch (*modules) {
1915  case '=':
1916  isLevel = true;
1917  isMod = false;
1918  break;
1919  case ',':
1920  isLevel = false;
1921  isMod = true;
1922  if (!ss.str().empty() && level != -1) {
1923  insert(ss, static_cast<base::type::VerboseLevel>(level));
1924  ss.str(std::string(""));
1925  level = -1;
1926  }
1927  break;
1928  default:
1929  if (isMod) {
1930  ss << *modules;
1931  } else if (isLevel) {
1932  if (isdigit(*modules)) {
1933  level = static_cast<base::type::VerboseLevel>(*modules) - 48;
1934  }
1935  }
1936  break;
1937  }
1938  }
1939  if (!ss.str().empty() && level != -1) {
1940  insert(ss, static_cast<base::type::VerboseLevel>(level));
1941  }
1942 }
