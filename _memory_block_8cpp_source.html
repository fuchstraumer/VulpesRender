21  constexpr static inline bool CheckBufferImageGranularityConflict(SuballocationType type_a, SuballocationType type_b) {
22  if (type_a > type_b) {
23  std::swap(type_a, type_b);
24  }
25 
26  switch (type_a) {
27  case SuballocationType::Free:
28  return false;
29  case SuballocationType::Unknown:
30  // best be conservative and play it safe: return true
31  return true;
32  case SuballocationType::Buffer:
33  // unknown return is playing it safe again, optimal return is because optimal tiling and linear buffers don't mix
34  return type_b == SuballocationType::ImageUnknown || type_b == SuballocationType::ImageOptimal;
35  case SuballocationType::ImageUnknown:

37  case SuballocationType::ImageLinear:
38  return type_b == SuballocationType::ImageOptimal;
39  case SuballocationType::ImageOptimal:
40  return false;
41  default:

164  bool prev_entry_free = false;

167  const Suballocation& curr = *iter;
168 
169  // if offset of current suballoc is wrong, then this allocation is invalid
170  if (curr.Offset != calculated_offset) {
171  return ValidationCode::INCORRECT_SUBALLOC_OFFSET;
172  }
173 
174  // two adjacent free entries is invalid: should be merged.
175  const bool curr_free = (curr.Type == SuballocationType::Free);
176  if (curr_free && prev_entry_free) {
177  return ValidationCode::NEED_MERGE_SUBALLOCS;
178  }
179 
180  // update prev free status
181  prev_entry_free = curr_free;
182 
183  if (curr_free) {
184  calculated_free_size += curr.Size;
185  ++calculated_free_suballocs;
186  if (curr.Size >= MinSuballocationSizeToRegister) {
187  ++num_suballocs_to_register; // this suballocation should be registered in the free list.
188  }
189  }
190 
191  calculated_offset += curr.Size;
192  }
193 
194  // Number of free suballocations in objects list doesn't match what our calculated value says we should have
195  if (availSuballocations.size() != num_suballocs_to_register) {
196  return ValidationCode::FREE_SUBALLOC_COUNT_MISMATCH;
197  }
198

200  for (size_t i = 0; i < availSuballocations.size(); ++i) {
201  auto curr_iter = availSuballocations[i];
202 
203  // non-free suballoc in free list
204  if (curr_iter->Type != SuballocationType::Free) {
205  return ValidationCode::USED_SUBALLOC_IN_FREE_LIST;
206  }
207 
208  // sorting of free list is incorrect
209  if (curr_iter->Size < last_entry_size) {
210  return ValidationCode::FREE_SUBALLOC_SORT_INCORRECT;
211  }
212

214  }
215 
216  if (calculated_offset != Size) {
217  return ValidationCode::FINAL_SIZE_MISMATCH;
218  }
219

221  return ValidationCode::FINAL_FREE_SIZE_MISMATCH;
222  }

273

279  }

282 
283  // Apply alignment
284  *dest_offset = AlignUp<VkDeviceSize>(*dest_offset, allocation_alignment);
285 
286 
287  // check previous suballocations for conflicts with buffer-image granularity, change alignment as needed
288  if (buffer_image_granularity > 1) {
289  bool conflict_found = false;
290  // iterate backwards, since we're checking previous suballoations for alignment conflicts
291  auto prev_suballoc_iter = dest_suballocation_location;
292  while (prev_suballoc_iter != Suballocations.cbegin()) {
293  --prev_suballoc_iter;
294  const Suballocation& prev_suballoc = *prev_suballoc_iter;
295  bool on_same_page = CheckBlocksOnSamePage(prev_suballoc.Offset, prev_suballoc.Size, *dest_offset, buffer_image_granularity);
296  if (on_same_page) {
297  conflict_found = CheckBufferImageGranularityConflict(prev_suballoc.Type, allocation_type);
298  if (conflict_found) {
299  LOG(WARNING) << "A buffer-image granularity conflict was identified in suballocation " << std::to_string(reinterpret_cast<size_t>(&prev_suballoc));
300  break;
301  }
302  }
303  else {
304  break;
305  }
306  }
307  if (conflict_found) {
308  // align up by a page size to get off the current page and remove the conflict.
309  *dest_offset = AlignUp<VkDeviceSize>(*dest_offset, buffer_image_granularity);
310  }
311  }
312 
313  // calculate padding at beginning from offset
314  const VkDeviceSize padding_begin = *dest_offset - suballoc.Offset;
315 
316  // calculate required padding at end, assuming current suballoc isn't at end of memory object
317  auto next_iter = dest_suballocation_location;
318  ++next_iter;
319  const VkDeviceSize padding_end = (next_iter != Suballocations.cend()) ? DEBUG_PADDING : 0;
320 
321  // Can't allocate if padding at begin and end is greater than requested size.
322  if (padding_begin + padding_end + allocation_size > suballoc.Size) {
323  LOG(WARNING) << "Suballocation verification failed as required padding for alignment + required size is greater than available space.";
324  return false;
325  }
326 
327  // We checked previous allocations for conflicts: now, we'll check next suballocations
328  if(buffer_image_granularity > 1) {
329  auto next_suballoc_iter = dest_suballocation_location;
330  ++next_suballoc_iter;
331  while (next_suballoc_iter != Suballocations.cend()) {
332  const auto& next_suballoc = *next_suballoc_iter;
333  bool on_same_page = CheckBlocksOnSamePage(*dest_offset, allocation_size, next_suballoc.Offset, buffer_image_granularity);
334  if (on_same_page) {
335  if (CheckBufferImageGranularityConflict(allocation_type, next_suballoc.Type)) {
336  LOG(WARNING) << "Suballocation verification failed as there were too many buffer-image granularity conflicts.";
337  return false;
338  }
339  }
340  else {
341  break;
342  }
343  ++next_suballoc_iter;
344  }
345  }
346 
347  return true;

379  }

382  if (padding_begin) {

384  auto next_iter = request.FreeSuballocation;
385  {

387  insertFreeSuballocation(insert_iter);
