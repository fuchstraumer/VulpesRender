85  uint32_t AllocatorImpl::findMemoryTypeIdx(const VkMemoryRequirements& mem_reqs, const AllocationRequirements & details) const noexcept {
86  auto req_flags = details.requiredFlags;
87  auto preferred_flags = details.preferredFlags;
88  if (req_flags == 0) {
89  assert(preferred_flags != VkMemoryPropertyFlagBits(0));
90  req_flags = preferred_flags;
91  }
92 
93  if (preferred_flags == 0) {
94  preferred_flags = req_flags;
95  }
96 
97  uint32_t min_cost = std::numeric_limits<uint32_t>::max();
98  uint32_t result_idx = std::numeric_limits<uint32_t>::max();
99  // preferred_flags, if not zero, must be a subset of req_flags
100  for (uint32_t type_idx = 0, memory_type_bit = 1; type_idx < GetMemoryTypeCount(); ++type_idx, memory_type_bit <<= 1) {
101  // memory type of idx is acceptable according to mem_reqs
102  if ((memory_type_bit & mem_reqs.memoryTypeBits) != 0) {
103  const VkMemoryPropertyFlags& curr_flags = deviceMemoryProperties.memoryTypes[type_idx].propertyFlags;
104  // current type contains required flags.
105  if ((req_flags & ~curr_flags) == 0) {
106  // calculate the cost of the memory type as the number of bits from preferred_flags
107  // not present in current type at type_idx.
108  uint32_t cost = countBitsSet(preferred_flags & ~req_flags);
109  if (cost < min_cost) {
110  result_idx = type_idx;
111  // ideal memory type, return it.
112  if (cost == 0) {
113  return result_idx;
114  }
115  min_cost = cost;
116  }
117  }
118 
119  }
120  }
121  // didn't find zero "cost" idx, but return it.
122  // this means that any methods that call this particular type finding method should handle the "exception"
123  // of invalid indices themselves.
124  return result_idx;
125  }
