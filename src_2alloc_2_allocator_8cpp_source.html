255  uint32_t Allocator::findMemoryTypeIdx(const VkMemoryRequirements& mem_reqs, const AllocationRequirements & details) const noexcept {
256  auto req_flags = details.requiredFlags;
257  auto preferred_flags = details.preferredFlags;
258  if (req_flags == 0) {
259  assert(preferred_flags != VkMemoryPropertyFlagBits(0));
260  req_flags = preferred_flags;
261  }
262 
263  if (preferred_flags == 0) {
264  preferred_flags = req_flags;
265  }
266 
267  uint32_t min_cost = std::numeric_limits<uint32_t>::max();
268  uint32_t result_idx = std::numeric_limits<uint32_t>::max();
269  // preferred_flags, if not zero, must be a subset of req_flags
270  for (uint32_t type_idx = 0, memory_type_bit = 1; type_idx < GetMemoryTypeCount(); ++type_idx, memory_type_bit <<= 1) {
271  // memory type of idx is acceptable according to mem_reqs
272  if ((memory_type_bit & mem_reqs.memoryTypeBits) != 0) {
273  const VkMemoryPropertyFlags& curr_flags = deviceMemoryProperties.memoryTypes[type_idx].propertyFlags;
274  // current type contains required flags.
275  if ((req_flags & ~curr_flags) == 0) {
276  // calculate the cost of the memory type as the number of bits from preferred_flags
277  // not present in current type at type_idx.
278  uint32_t cost = countBitsSet(preferred_flags & ~req_flags);
279  if (cost < min_cost) {
280  result_idx = type_idx;
281  // ideal memory type, return it.
282  if (cost == 0) {
283  return result_idx;
284  }
285  min_cost = cost;
286  }
287  }
288 
289  }
290  }
291  // didn't find zero "cost" idx, but return it.
292  // this means that any methods that call this particular type finding method should handle the "exception"
293  // of invalid indices themselves.
294  return result_idx;
295  }

301  // If given item is bigger than our preferred block size, we give it its own special allocation (using a single device memory object for this).
302  const bool private_memory = (alloc_details.prefersDedicatedKHR || alloc_details.requiresDedicatedKHR) || (memory_reqs.size > preferredBlockSize / 2);
303 
304  if (private_memory) {
305  if (AllocationRequirements::noNewAllocations) {
306  LOG(ERROR) << "Disabled creation of allocations outside of block-vectors and subregion binding, but tried to allocate a unique/private/dedicated allocation!";
307  throw std::domain_error("Required options for this type of allocation not enabled.");
308  }
309  else {
310  LOG_IF((alloc_details.prefersDedicatedKHR || alloc_details.requiresDedicatedKHR),INFO) << "Allocating dedicated memory object using extensions...";
311  return allocatePrivateMemory(memory_reqs.size, memory_type_idx, dest_allocation);
312  }
313  }
314  else {
315  const auto rec_alloc_bin = GetAllocSize(memory_reqs.size);
316 
317  auto& alloc_collection = allocations.at(rec_alloc_bin)[memory_type_idx];

321  SuballocationRequest request;
322  const auto& block = *iter;
323  if (block->RequestSuballocation(GetBufferImageGranularity(), memory_reqs.size, memory_reqs.alignment, type, &request)) {
324  if (block->Empty()) {

326  }
327 
328  block->Allocate(request, type, memory_reqs.size);
329  dest_allocation.Init(block.get(), request.Offset, memory_reqs.alignment, memory_reqs.size);
330

332  ValidationCode result_code = block->Validate();
333  if (result_code != ValidationCode::VALIDATION_PASSED) {
334  std::stringstream ss;
335  ss << "Validation of new allocation failed with reason: " << result_code;
336  const std::string fail_str{ ss.str() };
337  LOG(ERROR) << fail_str.c_str();
338  throw std::runtime_error(fail_str.c_str());
339  }
340  }
341  // only log for larger allocations of at least half a mb: otherwise, log gets clogged with updates
342  LOG_IF((static_cast<float>(memory_reqs.size) / 1.0e6f) > 0.5f, INFO) << "Successfully allocated by binding to suballocation with size of " << std::to_string(memory_reqs.size / 1e6) << "mb at offset " << std::to_string(request.Offset);

344  }
345  }
346

348  if (AllocationRequirements::noNewAllocations) {
349  LOG(ERROR) << "All available allocations full or invalid, and requested allocation not allowed to be private!";
350  return VK_ERROR_OUT_OF_DEVICE_MEMORY;
351  }
352  else {

354

358  break;
359  case AllocationSize::MEDIUM:
360  alloc_info.allocationSize = DefaultMediumHeapBlockSize;
361  break;
362  case AllocationSize::SMALL:
363  alloc_info.allocationSize = DefaultSmallHeapBlockSize;
364  break;

367  }

370  VkResult result = vkAllocateMemory(parent->vkHandle(), &alloc_info, nullptr, &new_memory);
371  assert(result != VK_ERROR_OUT_OF_DEVICE_MEMORY); // make sure we're not over-allocating and using all device memory.
372  if (result != VK_SUCCESS) {
373  // halve allocation size
374  alloc_info.allocationSize /= 2;
375  if (alloc_info.allocationSize >= memory_reqs.size) {
376  result = vkAllocateMemory(parent->vkHandle(), &alloc_info, nullptr, &new_memory);
377  if (result != VK_SUCCESS) {
378  alloc_info.allocationSize /= 2;
379  if (alloc_info.allocationSize >= memory_reqs.size) {
380  result = vkAllocateMemory(parent->vkHandle(), &alloc_info, nullptr, &new_memory);
381  }
382  }
383  }
384  }
385  // if still not allocated, try allocating private memory (if allowed)
386  if (result != VK_SUCCESS && (private_memory)) {
387  result = allocatePrivateMemory(memory_reqs.size, memory_type_idx, dest_allocation);
388  if (result == VK_SUCCESS) {
389  LOG(INFO) << "Allocation of memory succeeded";

391  }

395  }
