366  void Device::checkExtensions(std::vector<const char*>& extensions, bool throw_on_error) const {
367  uint32_t queried_count = 0;
368  vkEnumerateDeviceExtensionProperties(parent->vkHandle(), nullptr, &queried_count, nullptr);
369  std::vector<VkExtensionProperties> queried_extensions(queried_count);
370  vkEnumerateDeviceExtensionProperties(parent->vkHandle(), nullptr, &queried_count, queried_extensions.data());
371 
372  const auto has_extension = [queried_extensions](const char* name) {
373  const auto req_found = std::find_if(queried_extensions.cbegin(), queried_extensions.cend(), [name](const VkExtensionProperties& p) {
374  return strcmp(p.extensionName, name) == 0;
375  });
376  return req_found != queried_extensions.end();
377  };
378 
379  auto iter = extensions.begin();
380  while (iter != extensions.end()) {
381  if (!has_extension(*iter)) {
382  if (throw_on_error) {
383  LOG(ERROR) << "Current VkDevice does not support extension \"" << *iter << "\" that is required!";
384  throw std::runtime_error("Could not enable/use required extension for the logical device.");
385  }
386  else {
387  LOG(WARNING) << "Requested device extension with name \"" << *iter << "\" is not available, removing from list.";
388  extensions.erase(iter++);
389  }
390  }
391  else {
392  ++iter;
393  }
394  }
