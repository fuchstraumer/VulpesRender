461  void DeviceDataMembers::checkExtensions(std::vector<const char*>& extensions, bool throw_on_error) const {
462  uint32_t queried_count = 0;
463  vkEnumerateDeviceExtensionProperties(device->GetPhysicalDevice().vkHandle(), nullptr, &queried_count, nullptr);
464  std::vector<VkExtensionProperties> queried_extensions(queried_count);
465  vkEnumerateDeviceExtensionProperties(device->GetPhysicalDevice().vkHandle(), nullptr, &queried_count, queried_extensions.data());
466 
467  auto iter = std::remove_if(std::begin(extensions), std::end(extensions), [queried_extensions, throw_on_error](const char* name) {
468  const auto has_extension = [queried_extensions](const char* name) {
469  const auto req_found = std::find_if(queried_extensions.cbegin(), queried_extensions.cend(), [name](const VkExtensionProperties& p) {
470  return strcmp(p.extensionName, name) == 0;
471  });
472  return req_found != queried_extensions.end();
473  };
474 
475  if (!has_extension(name)) {
476  if (throw_on_error) {
477  LOG(ERROR) << "Current VkDevice does not support extension \"" << name << "\" that is required!";
478  throw std::runtime_error("Could not enable/use required extension for the logical device.");
479  }
480  else {
481  LOG(WARNING) << "Requested device extension with name \"" << name << "\" is not available, removing from list.";
482  }
483  return true;
484  }
485  else {
486  return false;
487  }
488  });
489 
490  extensions.erase(iter, extensions.end());
491  }
