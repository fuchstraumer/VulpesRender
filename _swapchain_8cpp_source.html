124  VkPresentModeKHR SwapchainImpl::SwapchainInfo::GetBestPresentMode() const{
125 
126  VkPresentModeKHR result = VK_PRESENT_MODE_FIFO_KHR;
127  for (const auto& mode : PresentModes) {
128  // Best mix of all modes.
129  if (mode == VK_PRESENT_MODE_MAILBOX_KHR) {
130  return VK_PRESENT_MODE_MAILBOX_KHR;

133  return VK_PRESENT_MODE_FIFO_KHR;

135  else if (mode == VK_PRESENT_MODE_IMMEDIATE_KHR) {
136  // FIFO not always supported by driver, in which case this is our best bet.

209  void SwapchainImpl::setParameters() {
210 
211  surfaceFormat = info.GetBestFormat();
212  colorFormat = surfaceFormat.format;
213  extent = info.ChooseSwapchainExtent(window);
214 
215  presentMode = info.GetBestPresentMode();
216  VkPresentModeKHR desiredMode = VK_PRESENT_MODE_BEGIN_RANGE_KHR;
217  switch (desiredSyncMode) {
218  case vertical_sync_mode::None:
219  desiredMode = VK_PRESENT_MODE_IMMEDIATE_KHR;
220  break;
221  case vertical_sync_mode::VerticalSync:
222  desiredMode = VK_PRESENT_MODE_FIFO_KHR;
223  break;
224  case vertical_sync_mode::VerticalSyncRelaxed:
225  desiredMode = VK_PRESENT_MODE_FIFO_RELAXED_KHR;
226  break;
227  case vertical_sync_mode::VerticalSyncMailbox:
228  desiredMode = VK_PRESENT_MODE_MAILBOX_KHR;
229  break;
230  default:
231  desiredMode = presentMode;
232  break;
233  }
234 
235  if (desiredMode != presentMode) {
236  auto& avail_modes = info.PresentModes;
237  auto iter = std::find(std::begin(avail_modes), std::end(avail_modes), desiredMode);
238  if (iter == std::cend(avail_modes)) {
239  LOG(WARNING) << "Desired vertical sync mode is " << getPresentModeStr(desiredMode) << "not available on current hardware!";
240  LOG(INFO) << "Falling back to supported present mode " << getPresentModeStr(presentMode);
241  }
242  else {
243  presentMode = *iter;
244  }
245  }
246 
247  // Create one more image than minspec to implement triple buffering (in hope we got mailbox present mode)
248  if (presentMode == VK_PRESENT_MODE_MAILBOX_KHR) {
249  imageCount = info.Capabilities.minImageCount + 1;
250  if (info.Capabilities.maxImageCount > 0 && imageCount > info.Capabilities.maxImageCount) {
251  imageCount = info.Capabilities.maxImageCount;
252  }
253  }
254  else if (presentMode == VK_PRESENT_MODE_FIFO_KHR || presentMode == VK_PRESENT_MODE_FIFO_RELAXED_KHR) {
255  imageCount = info.Capabilities.minImageCount + 1;
256  if (info.Capabilities.maxImageCount > 0 && imageCount > info.Capabilities.maxImageCount) {
257  imageCount = info.Capabilities.maxImageCount;
258  }
259  }
260  else {
261  imageCount = info.Capabilities.minImageCount;
262  }
